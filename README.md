  const blackScreen = document.getElementById('black-screen');
  const loadingScreen = document.getElementById('loading-screen');
  const welcomeText = document.getElementById('welcome-text');
  const progressBar = document.getElementById('progress');
  
  function randomChar() {
    const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
    return chars.charAt(Math.floor(Math.random() * chars.length));
  }

  // Pantalla negra con letras aleatorias
  let iteration = 0;
  const blackInterval = setInterval(() => {
    blackScreen.textContent = '';
    for(let i=0;i<100;i++){
      blackScreen.textContent += randomChar();
    }
    iteration++;
    if(iteration > 20){
      clearInterval(blackInterval);
      blackScreen.style.display='none';
      startLoading();
    }
  },50);

  // Barra + letras de bienvenida
  function startLoading(){
    loadingScreen.style.display='flex';
    let width=0;
    const targetText = 'WELCOME TO CAIMAN BANK';
    let displayed = Array(targetText.length).fill(' ');

    const progressInterval = setInterval(()=>{
      width += 1;
      progressBar.style.width = width + '%';

      for(let i=0;i<targetText.length;i++){
        if(displayed[i] !== targetText[i]){
          displayed[i] = Math.random() < width/100 ? targetText[i] : randomChar();
        }
      }
      welcomeText.textContent = displayed.join('');

      if(width>=100){
        clearInterval(progressInterval);
        loadingScreen.style.display='none';
        document.querySelector('main').style.display='block';
      }
    },92.7);
  }
</script>

    function randomInt(min, max) {
      return Math.floor(Math.random() * (max - min + 1)) + min;
    }

    function genVisaNumber() {
      const groups = [];
      for (let i = 0; i < 4; i++) {
        let block = randomInt(i === 0 ? 4000 : 0, i === 0 ? 4999 : 9999).toString().padStart(4,'0');
        groups.push(block);
      }
      return groups.join(' ');
    }
